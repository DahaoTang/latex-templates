\begin{lstlisting}[language=Python]
import math

def softmax(xs):
    m = max(xs)
    exps = [math.exp(x - m) for x in xs]
    s = sum(exps)
    return [e / s for e in exps]

print(softmax([1.0, 2.0, 3.0]))
